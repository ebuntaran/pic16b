Setting Up the Database

Most of this code is taken from the lecture “Working with Datasets”:

We first clean up the dataframes, then add them as databases using sqlite3.

import sqlite3

def prepare_df(df):
    df = df.set_index(keys=["ID", "Year"])
    df = df.stack()
    df = df.reset_index()
    df = df.rename(columns = {"level_2"  : "Month" , 0 : "Temp"})
    df["Month"] = df["Month"].str[5:].astype(int)
    df["Temp"]  = df["Temp"] / 100
    return(df)

with sqlite3.connect("temps.db") as conn: # create a database in current directory called temps.db
    temps_iter = pd.read_csv("temps.csv", chunksize = 100000)
    for i, df in enumerate(temps_iter):
        df = prepare_df(df)
        df.to_sql("temperatures", conn, if_exists = "replace" if i == 0 else "append", index = False)
    url = "station-metadata.csv"
    stations = pd.read_csv(url)
    stations.to_sql("stations", conn, if_exists = "replace", index=False)
    countries = pd.read_csv("https://raw.githubusercontent.com/mysociety/gaze/master/data/fips-10-4-to-iso-country-codes.csv")
    countries.to_sql("countries", conn, if_exists = "replace", index=False)

Querying the Database

The vast majority of the query_climate_database function is a single SQL query. We select the relevant columns and only take rows satisfying the given constraints. Since country and station data is not in the temperatures database, we need to do some merges in order to get all the relevant information into one table.

import inspect
from climate_database import query_climate_database
print(inspect.getsource(query_climate_database))
def query_climate_database(db_file, country, year_begin, year_end, month):
    """
    Query a climate database for temperature information

    Args:
        db_file: sqlite3 database file
        country (str): Country to take temperature data
        year_begin (int): Year to begin taking data
        year_end (int): Year to stop taking data
        month (int): Month to take temperature data
    
    Returns:
        df: Dataframe of temperature data at the provided country in the provided time range
    """

    cmd = \
    f"""
    SELECT S.name, S.latitude, S.longitude, C.Name, T.year, T.month, T.temp 
    FROM temperatures T
    LEFT JOIN stations S ON T.id = S.id
    INNER JOIN countries C ON SUBSTR(S.id,1,2) = C."FIPS 10-4"
    WHERE C.Name = "{country}" AND T.month = {month} AND T.year >= {year_begin} AND T.year <= {year_end}
    """
    with sqlite3.connect(db_file) as conn:
        df = pd.read_sql_query(cmd, conn)
        df.rename(columns={"Name": "Country"},inplace=True)
    return df

Plotting the Data

In order to first filter out stations that do not have enough observation data, we add a column showing the number of years each sation has been providing data for using transform, which we can then use to filter out those rows.

With transform, we can estimate the yearly temperature increase using linear regression. Finally, we use plotly’s scatter_mapbox function to make a plot of the data.

import plotly
from plotly import express as px
import numpy as np
import calendar
from sklearn.linear_model import LinearRegression

def temperature_coefficient_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    """
    Use a climate database to make a geographic scatterplot

    Args:
        db_file: sqlite3 database file
        country (str): Country to take temperature data
        year_begin (int): Year to begin taking data
        year_end (int): Year to stop taking data
        month (int): Month to take temperature data
        min_obs (int): Minimum number of years with recordings for stations
        **kwargs: Additional arguments to be passed into scatter_mapbox 
    Returns:
        fig: Geographic scatterplot of data
    """

    #collect data
    df = query_climate_database(db_file, country, year_begin, year_end, month)
    
    # we don't need the month and country columns anymore since month and country are arguments of the function
    df = df.drop(["Month","Country"], axis=1)

    # filter out stations with range of years of operation less than min_obs
    df["obs"] = df.groupby(["NAME", "LATITUDE", "LONGITUDE"])["Year"].transform(np.ptp)
    df = df[df["obs"]>min_obs]

    # defined in the lecture "Advanced Data Manipulation II": uses linear regression to calculate the estimated yearly increase
    def coef(data_group):
        x = data_group[["Year"]] # 2 brackets because X should be a df
        y = data_group["Temp"]   # 1 bracket because y should be a series
        LR = LinearRegression()
        LR.fit(x, y)
        return LR.coef_[0]

    # add a column for the yearly increase
    df = df.groupby(["NAME", "LATITUDE", "LONGITUDE"]).apply(coef)
    df = df.reset_index()
    df[0] = df[0].round(4)
    df.rename(columns={0: "Estimated Yearly Increase (°C)"},inplace=True)

    # create the plot with all the gathered information
    fig = px.scatter_mapbox(df, 
                            title = f"Estimates of yearly increase in temperature in {calendar.month_name[month]}<br>for stations in {country}, years {year_begin} - {year_end}",
                            hover_name = "NAME",
                            lat = "LATITUDE",
                            lon = "LONGITUDE", 
                            color = "Estimated Yearly Increase (°C)",
                            **kwargs)
    fig.update_coloraxes(cmid=0)
    return fig

Plotting Variation in Yearly Increase

We can see by the color scale that there can be significant variation of yearly increase within a country, but exactly how much is there? We can use the same data to make a box plot of yearly increases throughout a country, again using plotly.

import pandas as pd

def temperature_box_plot(db_file, countries, year_begin, year_end, month, min_obs):
    """
    Use a climate database to make a box plot

    Args:
        db_file: sqlite3 database file
        countries (list): Countries to take temperature data
        year_begin (int): Year to begin taking data
        year_end (int): Year to stop taking data
        month (int): Month to take temperature data
        min_obs (int): Minimum number of years with recordings for stations
    Returns:
        fig: Box plot of data
    """
    df = pd.DataFrame()
    #collect data
    for country in countries:
        df = pd.concat([df,query_climate_database(db_file, country, year_begin, year_end, month)], ignore_index = True)
    
    # we don't need the month column anymore since month is an argument of the function
    df = df.drop(["Month"], axis=1)

    # filter out stations with range of years of operation less than min_obs
    df["obs"] = df.groupby(["NAME", "LATITUDE", "LONGITUDE", "Country"])["Year"].transform(np.ptp)
    df = df[df["obs"]>min_obs]

    # defined in the lecture "Advanced Data Manipulation II": uses linear regression to calculate the estimated yearly increase
    def coef(data_group):
        x = data_group[["Year"]] # 2 brackets because X should be a df
        y = data_group["Temp"]   # 1 bracket because y should be a series
        LR = LinearRegression()
        LR.fit(x, y)
        return LR.coef_[0]

    # add a column for the yearly increase
    df = df.groupby(["NAME", "LATITUDE", "LONGITUDE", "Country"]).apply(coef)
    df = df.reset_index()
    df[0] = df[0].round(4)
    df.rename(columns={0: "Estimated Yearly Increase (°C)"},inplace=True)

    # create the plot with all the gathered information
    fig = px.box(df, "Estimated Yearly Increase (°C)", facet_col="Country",
                 title=f"Variation in yearly increase in temperature in {calendar.month_name[month]}<br>for stations in {country}, years {year_begin} - {year_end}",)
    return fig
fig3 = temperature_box_plot("temps.db", ["India","China","France"], 1980, 2020, 1, min_obs = 10)
fig3.show()

Is Yearly Increase Correlated With Latitude?

We’ve seen that yearly increase can vary significantly within a country, so could this be tied to the latitude of the stations?

To answer this question, we first need another SQL querying function.

query_climate_database_long accesses the same information as query_climate_database, but it instead takes stations that are within tol degrees of a given longitude (long).

from climate_database import query_climate_database_long
import inspect
print(inspect.getsource(query_climate_database_long))
def query_climate_database_long(db_file, long, tol, year_begin, year_end, month):
    """
    Query a climate database for temperature information

    Args:
        db_file: sqlite3 database file
        long (int): Longitude to take data from
        tol (int): Tolerance for longitude value
        year_begin (int): Year to begin taking data
        year_end (int): Year to stop taking data
        month (int): Month to take temperature data
    
    Returns:
        df: Dataframe of temperature data near the provided longitude in the provided time range
    """

    cmd = \
    f"""
    SELECT S.name, S.latitude, S.longitude, T.year, T.month, T.temp 
    FROM temperatures T
    LEFT JOIN stations S ON T.id = S.id
    WHERE T.month = {month} AND T.year >= {year_begin} AND T.year <= {year_end} AND S.longitude < {long+tol} AND S.longitude > {long-tol}
    """
    with sqlite3.connect(db_file) as conn:
        df = pd.read_sql_query(cmd, conn)
    return df

We can now use this function to gather data and plot it, again using plotly.
